  • Include the library interface fractal_color.h:
    #inlude "include/fractal_color.h"
  • Create a handle to a color matrix of specified width and height:
    HCMATRIX hCmatrix = fractal_cmatrix_create(800, 800);

  • Specify the desired fractal properties:

    float c_real = -0.7835f;
    float c_imag = -0.2321f;
    float R = 2.0f;
    int height = 8*100;
    int width = 8*240;
    float aspect_ratio = (float)width/height;
    float x_start = -R;
    float x_end = R;
    float y_start = x_start/aspect_ratio;
    float y_end = x_end/aspect_ratio;
    int max_iterations = 1000;
    enum FC_Mode mode = FC_MODE_JULIA;

    .frac = fractal,
    .mode = mode,
    .color = color,
    .c_real = c_real,
    .c_imag = c_imag,
    .R = R,
    .max_iterations = max_iterations,
